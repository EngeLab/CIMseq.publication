## Joining, by = "sample"

Precision: within any given class (defined by unsupervised classification) what fraction of times is a sample of that class devonvoluted correctly.

data %>%
  mutate(correct = cells == class) %>%
  group_by(class) %>%
  summarize(precision = sum(correct) / n()) %>%
  arrange(precision) %>%
  print(n = nrow(.))

Mean precision per class

data %>%
  mutate(correct = cells == class) %>%
  group_by(class) %>%
  summarize(precision = sum(correct) / n()) %>% 
  pull(precision) %>% 
  mean()
